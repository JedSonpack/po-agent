# 顶部 import 加 pytest；s08 的 12 个测试 sed 复制（s08_context_compact→s09_memory）

class SpyMemory:
    def __init__(self):
        self.calls = []
        self.inject = ""
        self.section = ""
    def load_memories(self, messages): self.calls.append("load"); return self.inject
    def build_index_section(self): return self.section
    def extract_memories(self, msgs): self.calls.append(("extract", msgs))
    def consolidate_memories(self): self.calls.append("consolidate")


def test_memory_loaded_at_loop_start():
    client = FakeClient([make_response([text_block("done")], "end_turn")])
    spy = SpyMemory()
    agent_loop(client=client, model="m", system="s", tools=[], messages=[{"role":"user","content":"x"}],
               run_tool=lambda n,i:"OUT", trigger=lambda ev,*a:None, memory=spy)
    assert "load" in spy.calls


def test_memory_extracted_and_consolidated_at_turn_end():
    client = FakeClient([make_response([text_block("done")], "end_turn")])
    spy = SpyMemory()
    agent_loop(client=client, model="m", system="s", tools=[], messages=[{"role":"user","content":"x"}],
               run_tool=lambda n,i:"OUT", trigger=lambda ev,*a:None, memory=spy)
    assert any(isinstance(c, tuple) and c[0]=="extract" for c in spy.calls)
    assert "consolidate" in spy.calls


def test_memory_index_appended_to_system():
    client = FakeClient([make_response([text_block("done")], "end_turn")])
    captured = {}
    def capture_create(**kw): captured["system"] = kw.get("system"); return make_response([text_block("done")], "end_turn")
    client2 = type("C",(),{"messages":property(lambda s:s), "create":capture_create})()
    spy = SpyMemory(); spy.section = "\n\nMemories available:\n- [X](x.md) — dx"
    agent_loop(client=client2, model="m", system="BASE", tools=[], messages=[{"role":"user","content":"x"}],
               run_tool=lambda n,i:"OUT", trigger=lambda ev,*a:None, memory=spy)
    assert "BASE" in captured["system"]
    assert "Memories available" in captured["system"]


def test_memory_injected_into_user_turn():
    client = FakeClient([make_response([text_block("done")], "end_turn")])
    captured = {}
    def capture_create(**kw): captured["messages"] = kw.get("messages"); return make_response([text_block("done")], "end_turn")
    client2 = type("C",(),{"messages":property(lambda s:s), "create":capture_create})()
    spy = SpyMemory(); spy.inject = "MEMCONTENT"
    agent_loop(client=client2, model="m", system="s", tools=[], messages=[{"role":"user","content":"hello"}],
               run_tool=lambda n,i:"OUT", trigger=lambda ev,*a:None, memory=spy)
    # 第一条 user 消息内容应含 MEMCONTENT + hello
    assert "MEMCONTENT" in str(captured["messages"][0]["content"])
    assert "hello" in str(captured["messages"][0]["content"])
